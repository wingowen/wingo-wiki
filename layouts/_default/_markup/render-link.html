{{/* Render hook for links and wikilinks */}} {{- $destination := .Destination -}} {{- $text := .Text -}} {{- $title := .Title -}} {{- /* Try to find the page for internal links */ -}} {{- $page := "" -}} {{- $isInternal := not (or (hasPrefix $destination "http://") (hasPrefix $destination "https://") (hasPrefix $destination "#")) -}} {{- if $isInternal -}} {{- /* Clean the destination - remove .md extension if present */ -}} {{- $cleanDest := $destination -}} {{- if strings.HasSuffix $cleanDest ".md" -}} {{- $cleanDest = strings.TrimSuffix ".md" $cleanDest -}} {{- end -}} {{- if strings.HasSuffix $cleanDest "/" -}} {{- $cleanDest = strings.TrimSuffix "/" $cleanDest -}} {{- end -}} {{- /* Try multiple patterns to find the page */ -}} {{- $page = site.RegularPages.GetMatch (print $cleanDest ".md") -}} {{- if not $page -}} {{- $page = site.RegularPages.GetMatch (print "**/" $cleanDest ".md") -}} {{- end -}} {{- if not $page -}} {{- $page = site.RegularPages.GetMatch $cleanDest -}} {{- end -}} {{- if not $page -}} {{- $page = site.RegularPages.GetMatch (print "**/" $cleanDest) -}} {{- end -}} {{- /* If found, use the page's RelPermalink */ -}} {{- if $page -}} {{ $text }} {{- else -}} {{- /* If not found, render as plain text to avoid 404 */ -}} {{ $text }} {{- end -}} {{- else -}} {{- /* Render regular external or anchor links */ -}} {{ $text }} {{- end -}}
