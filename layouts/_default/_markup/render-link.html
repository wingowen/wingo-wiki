{{/* Render hook for wikilinks */}} {{- $destination := .Destination -}} {{- $text := .Text -}} {{- $title := .Title -}} {{- /* Check if this is a wikilink */ -}} {{- if eq $destination "wikilink" -}} {{- /* Parse the link target and display text */ -}} {{- $target := $title -}} {{- $display := $text -}} {{- /* Find the page */ -}} {{- $page := site.RegularPages.GetMatch $target -}} {{- if not $page -}} {{- $page = site.RegularPages.GetMatch (print $target ".md") -}} {{- end -}} {{- if not $page -}} {{- $page = site.RegularPages.GetMatch (print "**/" $target) -}} {{- end -}} {{- if not $page -}} {{- $page = site.RegularPages.GetMatch (print "**/" $target ".md") -}} {{- end -}} {{- /* Render the link */ -}} {{- if $page -}} {{ $display }} {{- else -}} {{ $display }} {{- end -}} {{- else -}} {{- /* Render regular links */ -}} {{ $text }} {{- end -}}
